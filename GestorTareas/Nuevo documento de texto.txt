import React, { useState } from "react";
import { useCardContext } from "./CardContext";  // Importamos el hook

export const Prueva = () => {
  const { cards, addCard } = useCardContext();  // Usamos el contexto

  const [inputValue, setInputValue] = useState('');  // Estado para manejar el valor del input

  // Función que se ejecuta al hacer click en el botón
  const handleAddCard = () => {
    addCard(inputValue);  // Usamos la función `addCard` del contexto
    setInputValue('');     // Limpiamos el input después de agregar la tarjeta
  };

  return (
    <div>
      <input
        type="text"
        value={inputValue}
        onChange={(e) => setInputValue(e.target.value)}
        placeholder="Nombre usuario"
      />

      <button onClick={handleAddCard}>Agregar tarjeta</button>

      <div>
        {cards.map((card, index) => (
          <div key={index} className="card">
            <h3>{card.title}</h3>
            <p>{card.description}</p>
          </div>
        ))}
      </div>
    </div>
  );
};
